def report_metrics 
  metrics.each do |k,v|
    if v.first
      csvstr = $mod.generate(:force_quotes=>true) do |csv|
        csv << [k.to_s] + v.first[0].keys + [:val]
        v.each do |row|
          csv << [nil] + row[0].values + [row[1]]
        end
      end
      puts csvstr
    end
  end
end
