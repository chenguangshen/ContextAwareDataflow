  def ==(o)
    if o.class == self.class
      return oldeq(o)
    elsif o.class == Array
      begin
        self.each_with_index do |el, i|
          if el != o[i]
            return false
          end
        end
        return true
      rescue StandardError
        return false
      end
    end
    false
  end
