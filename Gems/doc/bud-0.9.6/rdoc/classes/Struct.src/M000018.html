  def <=>(o)
    if o.class == self.class
      self.each_with_index do |e, i|
        cmp = e <=> o[i]
        return cmp if cmp != 0
      end
      return 0
    elsif o.nil?
      return -1
    else
      raise "Comparison (<=>) between #{o.class} and #{self.class} not implemented"
    end
  end
