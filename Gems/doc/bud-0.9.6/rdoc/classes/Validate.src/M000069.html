  def collapse(left, right)
    return [right] if left == 'Bot'
    return [left] if right == 'Bot'
    return [left] if left == right
    return ['D'] if left == 'D' or right == 'D'
    # CALM
    return ['D'] if left == 'A' and right =~ /N/
    # sometimes we cannot reduce
    return [left, right]
  end
