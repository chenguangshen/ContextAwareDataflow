  def merge_to_storage(v)
    m = do_merge(current_value, v)
    if m != current_value
      @storage = m
      @rescan_on_delta.each do |e|
        if e.kind_of? Bud::ScannerElement
          e.force_rescan = true
        else
          e.rescan = true
        end
      end
      return true
    else
      return false
    end
  end
