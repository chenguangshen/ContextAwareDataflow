  def self.morph(name, &block)
    if mfuncs.include?(name) || @@global_mfuncs.include?(name)
      raise Bud::CompileError, "#{name} declared as both monotone and morph"
    end
    @morphs ||= Set.new
    @morphs << name
    @@global_morphs << name
    define_method(name, &block)
  end
