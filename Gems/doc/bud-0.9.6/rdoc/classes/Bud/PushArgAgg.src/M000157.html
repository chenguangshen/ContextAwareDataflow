    def initialize(elem_name, bud_instance, collection_name, keys_in, aggpairs_in, schema_in, &blk)

        raise Bud::Error, "multiple aggpairs #{aggpairs_in.map{|a| a.class.name}} in ArgAgg; only one allowed"

    end
