    def invalidate_cache
      raise Bud::Error if @rhs_rcvd     # sanity check; should already be reset

      if @lhs.rescan
        puts "#{tabname} rel:#{@lhs.tabname} invalidated" if $BUD_DEBUG
        @hash_tables[0] = {}
      end
      if @rhs.rescan
        puts "#{tabname} rel:#{@rhs.tabname} invalidated" if $BUD_DEBUG
        @hash_tables[1] = {}
      end
    end
