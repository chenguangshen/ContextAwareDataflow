    def pairs(*preds, &blk)
      @origpreds = preds
      setup_preds(preds) unless preds.empty?
      # given new preds, the state for the join will be different.  set it up again.
      setup_state if self.class <= Bud::PushSHJoin
      set_block(&blk) if blk
      self
    end
