    def replay_join
      @rescan = false
      a, b = @hash_tables
      return if a.empty? or b.empty?

      if a.size < b.size
        a.each_pair do |key, items|
          the_matches = b[key]
          unless the_matches.nil?
            items.each do |item|
              process_matches(item, the_matches, 0)
            end
          end
        end
      else
        b.each_pair do |key, items|
          the_matches = a[key]
          unless the_matches.nil?
            items.each do |item|
              process_matches(item, the_matches, 1)
            end
          end
        end
      end
    end
