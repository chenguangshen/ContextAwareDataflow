    def flush
      # Don't emit fresh output unless a rescan is needed
      return unless @rescan
      @rescan = false

      @groups.each do |key, group_state|
        rv = key.clone
        @aggpairs.each_with_index do |ap, agg_ix|
          rv << ap[0].final(group_state[agg_ix])
        end
        push_out(rv)
      end
    end
