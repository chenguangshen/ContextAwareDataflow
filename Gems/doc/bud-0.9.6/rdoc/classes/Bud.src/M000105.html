  def run_bg
    start

    schedule_and_wait do
      if @running_async
        raise Bud::Error, "run_bg called on already-running Bud instance"
      end
      @running_async = true

      # Consume any events received while we weren't running async
      tick_internal
    end

    @rtracer.sleep if options[:rtrace]
  end
