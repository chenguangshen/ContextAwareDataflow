  def interfaces(direction, collections)
    mode = case direction
      when :input then true
      when :output then false
    else
      raise Bud::CompileError, "unrecognized interface type #{direction}"
    end
    collections.each do |tab|
      t_provides << [tab.to_s, mode]
    end 
  end
