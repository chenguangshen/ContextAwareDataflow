  def do_register_callback(tbl_name, &blk)
    unless @tables.has_key? tbl_name
      raise Bud::Error, "no such collection: #{tbl_name}"
    end

    raise Bud::Error if @callbacks.has_key? @callback_id
    @callbacks[@callback_id] = [tbl_name, blk]
    cb_id = @callback_id
    @callback_id += 1
    return cb_id
  end
