  def self.setup_history
    begin
      if File::exists?(@@histfile)
        lines = IO::readlines(@@histfile).collect { |line| line.chomp }
        Readline::HISTORY.push(*lines)
      end
    rescue Exception
      puts "Error when configuring permanent history: #{$!}"
    end
  end
