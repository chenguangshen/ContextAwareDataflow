  def initialize(mod)
    @report = nil
    @mod = Object.const_get(mod)
    if @mod.class == Class
      nc = new_class_from_class(@mod)
    elsif @mod.class == Module
      nc = new_class(@mod)
    else
      raise "#{mod} neither class nor module"
    end
    @f = nc.new
    @f.tick
  end
