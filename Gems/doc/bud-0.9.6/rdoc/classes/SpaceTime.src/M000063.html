  def finish(file, fmt=nil)
    @edges.each_pair do |k, v|
      lbl = v[3] > 1 ? "#{v[2]}(#{v[3]})" : v[2]
      lbl ||= ""
      @g.add_edges(v[0], v[1], :label => lbl, :color => "red", :weight => 1)
    end
    if fmt.nil?
      @g.output(:svg => "#{file}.svg")
    else
      eval("@g.output(:#{fmt} => \"\#{file}.#{fmt}\")")
    end
  end
