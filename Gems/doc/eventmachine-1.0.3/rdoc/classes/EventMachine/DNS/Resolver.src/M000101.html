      def self.socket
        if !@socket || (@socket && @socket.error?)
          @socket = Socket.open

          @hosts  = {}
          IO.readlines('/etc/hosts').each do |line|
            next if line =~ /^#/
            addr, host = line.split(/\s+/)

            if @hosts[host]
              @hosts[host] << addr
            else
              @hosts[host] = [addr]
            end
          end
        end

        @socket
      end
