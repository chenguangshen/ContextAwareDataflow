        def headers_2_hash hdrs
          hash = {}
          hdrs.each {|h|
            if /\A([^\s:]+)\s*:\s*/ =~ h
              tail = $'.dup
              hash[ $1.downcase.gsub(/-/,"_").intern ] = tail
            end
          }
          hash
        end
