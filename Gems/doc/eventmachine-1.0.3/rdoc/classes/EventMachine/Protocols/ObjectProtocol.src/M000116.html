      def receive_data data
        (@buf ||= '') << data

        while @buf.size >= 4
          if @buf.size >= 4+(size=@buf.unpack('N').first)
            @buf.slice!(0,4)
            receive_object serializer.load(@buf.slice!(0,size))
          else
            break
          end
        end
      end
