Class EventMachine::PeriodicTimer
In: lib/em/timers.rb
Parent: Object

Creates a periodic timer

@example 

 n = 0
 timer = EventMachine::PeriodicTimer.new(5) do
   puts "the time is #{Time.now}"
   timer.cancel if (n+=1) > 5
 end
